const: 상수 사용

Arrow Functions

```js
function myFnc() {
    ...
}
    
const myFnc = () =>{ //인수가 한개면 괄호 생략 가능. 인수가 없거나 2개 이상이면 괄호 필수
    ...
    //this 가 변하지 않음 
}
    
const multiply = number => number*2; //return 명령줄 한개만 있으면 중괄호도 생략 가능
```
